จงหาผลลัพธ์ทั้งหมดที่เกิดจากคำสั่ง console.log

```js
console.log("1" > 4); // f
console.log("" <= 1); // t
console.log("apple" > "pineapple"); // f
console.log(undefined == null); // t
console.log(undefined === null); // f
console.log(NaN !== 2); // t
console.log(NaN == NaN); // f
console.log(NaN === NaN); // f
```
